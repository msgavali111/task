document.addEventListener('DOMContentLoaded', function() {
    var disclosureList = "Are Not FDIC ,Are Not FDIC Insured ,Are Not Bank Guaranteed,May Lose Value,Are Not Deposits,Are Not Insured by Any Federal Government Agency,Are Not a Condition to Any Banking Service or Activity,Disclosure one, Are Not FDIC Insured,Are Not Bank Guaranteed,May Lose Value,Are Not Deposits,Are Not Insured by Any Federal Government Agency,Are Not a Condition to Any Banking Service or Activity,Disclosure one";    
    buildDisclosureTable(disclosureList);
});
function buildDisclosureTable(disclosureList) {
    var items = disclosureList.split(',');

    var tbody = document.getElementById('document-list-tbody');

    for (var i = 0; i < items.length; i++) {
        if (i % 2 === 0) {
            var row = tbody.insertRow();
        }

        var cell = row.insertCell();

        var div = document.createElement('div');
        div.className = "td-div";

        var img = document.createElement('img');
        img.src = 'icons8-pdf-96.png'; // Set the path to your image
        img.alt = 'PDF Icon'; // Set an appropriate alt text

        var link = document.createElement('a');
        link.href = '#'; // Set the URL you want the link to point to
        link.textContent = items[i].trim();

        div.appendChild(img);
        div.appendChild(link);         
        cell.appendChild(div);
        
    }

}
