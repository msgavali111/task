document.addEventListener('DOMContentLoaded', function() {
    var component = document.querySelector('.disclosure-list');
    var disclosureList = component.getAttribute('data-disclosureList');
    buildDisclosureTable(disclosureList);
});

function buildDisclosureTable(disclosureList) {
    var items = disclosureList.split(',');

    var tbody = document.getElementById('disclosure-tbody');

    for (var i = 0; i < items.length; i++) {
        if (i % 3 === 0) {
            var row = tbody.insertRow();
        }

        var cell = row.insertCell();
        var span = document.createElement('span');
        span.textContent = items[i].trim();
        cell.appendChild(span);
    }
}
